1 לַמְנַצֵּ֬חַ לִבְנֵי־קֹ֬רַח מַשְׂכִּֽיל׃ 2 אֱלֹהִ֤ים ׀ בְּאָזְנֵ֬ינוּ שָׁמַ֗עְנוּ אֲבֹותֵ֥ינוּ סִפְּרוּ־לָ֑נוּ פֹּ֥עַל פָּעַ֥לְתָּ בִֽ֝ימֵיהֶ֗ם בִּ֣ימֵי קֶֽדֶם׃ 3 אַתָּ֤ה ׀ יָדְךָ֡ גֹּויִ֣ם הֹ֭ורַשְׁתָּ וַתִּטָּעֵ֑ם תָּרַ֥ע לְ֝אֻמִּ֗ים וַֽתְּשַׁלְּחֵֽם׃ 4 כִּ֤י לֹ֪א בְחַרְבָּ֡ם יָ֥רְשׁוּ אָ֗רֶץ וּזְרֹועָם֮ לֹא־הֹושִׁ֪יעָ֫ה לָּ֥מֹו כִּֽי־יְמִֽינְךָ֣ וּ֭זְרֹועֲךָ וְאֹ֥ור פָּנֶ֗יךָ כִּ֣י רְצִיתָֽם׃ 5 אַתָּה־ה֣וּא מַלְכִּ֣י אֱלֹהִ֑ים צַ֝וֵּ֗ה יְשׁוּעֹ֥ות יַעֲקֹֽב׃ 6 בְּ֭ךָ צָרֵ֣ינוּ נְנַגֵּ֑חַ בְּ֝שִׁמְךָ֗ נָב֥וּס קָמֵֽינוּ׃ 7 כִּ֤י לֹ֣א בְקַשְׁתִּ֣י אֶבְטָ֑ח וְ֝חַרְבִּ֗י לֹ֣א תֹושִׁיעֵֽנִי׃ 8 כִּ֣י הֹ֭ושַׁעְתָּנוּ מִצָּרֵ֑ינוּ וּמְשַׂנְאֵ֥ינוּ הֱבִישֹֽׁותָ׃ 9 בֵּֽ֭אלֹהִים הִלַּלְ֣נוּ כָל־הַיֹּ֑ום וְשִׁמְךָ֓ ׀ לְעֹולָ֖ם נֹודֶ֣ה סֶֽלָה׃ 10 אַף־זָ֭נַחְתָּ וַתַּכְלִימֵ֑נוּ וְלֹא־תֵ֝צֵ֗א בְּצִבְאֹותֵֽינוּ׃ 11 תְּשִׁיבֵ֣נוּ אָ֭חֹור מִנִּי־צָ֑ר וּ֝מְשַׂנְאֵ֗ינוּ שָׁ֣סוּ לָֽמֹו׃ 12 תִּ֭תְּנֵנוּ כְּצֹ֣אן מַאֲכָ֑ל וּ֝בַגֹּויִ֗ם זֵרִיתָֽנוּ׃ 13 תִּמְכֹּֽר־עַמְּךָ֥ בְלֹא־הֹ֑ון וְלֹ֥א־רִ֝בִּ֗יתָ בִּמְחִירֵיהֶֽם׃ 14 תְּשִׂימֵ֣נוּ חֶ֭רְפָּה לִשְׁכֵנֵ֑ינוּ לַ֥עַג וָ֝קֶ֗לֶס לִסְבִיבֹותֵֽינוּ׃ 15 תְּשִׂימֵ֣נוּ מָ֭שָׁל בַּגֹּויִ֑ם מְנֹֽוד־רֹ֝֗אשׁ בַּל־אֻמִּֽים׃ 16 כָּל־הַ֭יֹּום כְּלִמָּתִ֣י נֶגְדִּ֑י וּבֹ֖שֶׁת פָּנַ֣י כִּסָּֽתְנִי׃ 17 מִ֭קֹּול מְחָרֵ֣ף וּמְגַדֵּ֑ף מִפְּנֵ֥י אֹ֝ויֵ֗ב וּמִתְנַקֵּֽם׃ 18 כָּל־זֹ֣את בָּ֭אַתְנוּ וְלֹ֣א שְׁכַחֲנ֑וּךָ וְלֹֽא־שִׁ֝קַּ֗רְנוּ בִּבְרִיתֶֽךָ׃ 19 לֹא־נָסֹ֣וג אָחֹ֣ור לִבֵּ֑נוּ וַתֵּ֥ט אֲשֻׁרֵ֗ינוּ מִנִּ֥י אָרְחֶֽךָ׃ 20 כִּ֣י דִ֭כִּיתָנוּ בִּמְקֹ֣ום תַּנִּ֑ים וַתְּכַ֖ס עָלֵ֣ינוּ בְצַלְמָֽוֶת׃ 21 אִם־שָׁ֭כַחְנוּ שֵׁ֣ם אֱלֹהֵ֑ינוּ וַנִּפְרֹ֥שׂ כַּ֝פֵּ֗ינוּ לְאֵ֣ל זָֽר׃ 22 הֲלֹ֣א אֱ֭לֹהִים יַֽחֲקָר־זֹ֑את כִּֽי־ה֥וּא יֹ֝דֵ֗עַ תַּעֲלֻמֹ֥ות לֵֽב׃ 23 כִּֽי־עָ֭לֶיךָ הֹרַ֣גְנוּ כָל־הַיֹּ֑ום נֶ֝חְשַׁ֗בְנוּ כְּצֹ֣אן טִבְחָֽה׃ 24 ע֤וּרָה ׀ לָ֖מָּה תִישַׁ֥ן ׀ אֲדֹנָ֑י הָ֝קִ֗יצָה אַל־תִּזְנַ֥ח לָנֶֽצַח׃ 25 לָֽמָּה־פָנֶ֥יךָ תַסְתִּ֑יר תִּשְׁכַּ֖ח עָנְיֵ֣נוּ וְֽלַחֲצֵֽנוּ׃ 26 כִּ֤י שָׁ֣חָה לֶעָפָ֣ר נַפְשֵׁ֑נוּ דָּבְקָ֖ה לָאָ֣רֶץ בִּטְנֵֽנוּ׃ 27 ק֭וּמָֽה עֶזְרָ֣תָה לָּ֑נוּ וּ֝פְדֵ֗נוּ לְמַ֣עַן חַסְדֶּֽךָ׃
