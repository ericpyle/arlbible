1 فَٱطْرَحُوا كُلَّ خُبْثٍ وَكُلَّ مَكْرٍ وَٱلرِّيَاءَ وَٱلْحَسَدَ وَكُلَّ مَذَمَّةٍ✡ ص4‏:2؛ أف4‏:2، 25، 31؛ كو3‏:8؛ عب12‏:1؛ يع1‏:21؛ 5‏:9، 2 وَكَأَطْفَالٍ✡ ص1‏:23؛ مت18‏:3؛ مر10‏:15؛ رو6‏:4؛ 1كو14‏:20 مَوْلُودِينَ ٱلْآنَ، ٱشْتَهُوا ٱللَّبَنَ ٱلْعَقْلِيَّ✡ 1كو3‏:2؛ عب5‏:12، 13 ٱلْعَدِيمَ ٱلْغِشِّ لِكَيْ تَنْمُوا بِهِ، 3 إِنْ كُنْتُمْ قَدْ ذُقْتُمْ أَنَّ ٱلرَّبَّ صَالِحٌ✡ مز34‏:8؛ عب6‏:5* أو لطيف. انظر رو2‏:4؛ أف4‏:32.
الحجر الحي والشعب المختار
4 ٱلَّذِي إِذْ تَأْتُونَ إِلَيْهِ، حَجَرًا حَيًّا مَرْفُوضًا مِنَ ٱلنَّاسِ✡ مز118‏:22؛ مت21‏:42؛ أع4‏:11 ، وَلَكِنْ مُخْتَارٌ مِنَ ٱللهِ كَرِيمٌ، 5 كُونُوا أَنْتُمْ أَيْضًا مَبْنِيِّينَ -كَحِجَارَةٍ حَيَّةٍ✡ أف2‏:21، 22 - بَيْتًا رُوحِيًّا✡ عب3‏:6 ، كَهَنُوتًا مُقَدَّسًا✡ ع9؛ إش61‏:6؛ 66‏:21، لِتَقْدِيمِ ذَبَائِحَ رُوحِيَّةٍ✡ هو14‏:2؛ ملا1‏:11؛ رو12‏:1؛ عب13‏:15، 16 مَقْبُولَةٍ عِنْدَ ٱللهِ بِيَسُوعَ ٱلْمَسِيحِ✡ ص4‏:11؛ في4‏:18. 6 لِذَلِكَ يُتَضَمَّنُ أَيْضًا فِي ٱلْكِتَابِ✡ إش28‏:16؛ رو9‏:33 : «هَأَنَذَا أَضَعُ فِي صِهْيَوْنَ حَجَرَ زَاوِيَةٍ مُخْتَارًا كَرِيمًا، وَٱلَّذِي يُؤْمِنُ بِهِ لَنْ يُخْزَى». 7 فَلَكُمْ أَنْتُمُ ٱلَّذِينَ تُؤْمِنُونَ ٱلْكَرَامَةُ، وَأَمَّا لِلَّذِينَ لَا يُطِيعُونَ، «فَٱلْحَجَرُ ٱلَّذِي رَفَضَهُ ٱلْبَنَّاؤُونَ، هُوَ قَدْ صَارَ رَأْسَ ٱلزَّاوِيَةِ✡ مز118‏:22؛ مت21‏:42؛ أع4‏:11 » 8 وَ«حَجَرَ صَدْمَةٍ✡ إش8‏:14؛ لو2‏:34؛ رو9‏:33 وَصَخْرَةَ عَثْرَةٍ». ٱلَّذِينَ يَعْثُرُونَ✡ 1كو1‏:23 غَيْرَ طَائِعِينَ لِلْكَلِمَةِ، ٱلْأَمْرُ ٱلَّذِي جُعِلُوا لَهُ✡ خر9‏:16؛ رو9‏:22؛ 1تس5‏:9؛ يه4 . 9 وَأَمَّا أَنْتُمْ فَجِنْسٌ مُخْتَارٌ✡ ص1‏:1؛ تث10‏:15، وَكَهَنُوتٌ مُلُوكِيٌّ✡ خر19‏:5، 6؛ رؤ1‏:6؛ 5‏:10، أُمَّةٌ مُقَدَّسَةٌ✡ يو17‏:19؛ 1كو3‏:17؛ 2تي1‏:9، شَعْبُ ٱقْتِنَاءٍ✡ تث4‏:20؛ 7‏:6؛ 14‏:2؛ 26‏:18، 19؛ أع20‏:28؛ أف1‏:14؛ تي2‏:14 ، لِكَيْ تُخْبِرُوا بِفَضَائِلِ ٱلَّذِي دَعَاكُمْ مِنَ ٱلظُّلْمَةِ إِلَى نُورِهِ ٱلْعَجِيبِ✡ أع26‏:18؛ أف5‏:8؛ كو1‏:13؛ 1تس5‏:4، 5 . 10 ٱلَّذِينَ قَبْلًا لَمْ تَكُونُوا شَعْبًا، وَأَمَّا ٱلْآنَ فَأَنْتُمْ شَعْبُ ٱللهِ. ٱلَّذِينَ كُنْتُمْ غَيْرَ مَرْحُومِينَ، وَأَمَّا ٱلْآنَ فَمَرْحُومُونَ✡ هو1‏:9، 10؛ 2‏:23؛ رو9‏:25 .
الخضوع للسلطة
11 أَيُّهَا ٱلْأَحِبَّاءُ، أَطْلُبُ إِلَيْكُمْ كَغُرَبَاءَ وَنُزَلَاءَ✡ ص1‏:17؛ 1أخ29‏:15؛ مز39‏:12؛ 119‏:19؛ عب11‏:13، أَنْ تَمْتَنِعُوا† أو تبتعدوا عَنِ ٱلشَّهَوَاتِ ٱلْجَسَدِيَّةِ✡ رو13‏:14؛ غل5‏:16 ٱلَّتِي تُحَارِبُ ٱلنَّفْسَ✡ يع4‏:1 ، 12 وَأَنْ تَكُونَ سِيرَتُكُمْ بَيْنَ ٱلْأُمَمِ حَسَنَةً✡ ص3‏:16؛ رو12‏:17؛ 2كو8‏:21؛ في2‏:15؛ تي2‏:8، لِكَيْ يَكُونُوا، فِي مَا يَفْتَرُونَ عَلَيْكُمْ‡ أو يذمونكم كَفَاعِلِي شَرٍّ، يُمَجِّدُونَ ٱللهَ فِي يَوْمِ ٱلِٱفْتِقَادِ✡ لو19‏:44 ، مِنْ أَجْلِ أَعْمَالِكُمُ ٱلْحَسَنَةِ ٱلَّتِي يُلَاحِظُونَهَا✡ مت5‏:16 . 13 فَٱخْضَعُوا لِكُلِّ تَرْتِيبٍ بَشَرِيٍّ مِنْ أَجْلِ ٱلرَّبِّ✡ مت22‏:21؛ رو13‏:1؛ تي3‏:1 . إِنْ كَانَ لِلْمَلِكِ فَكَمَنْ هُوَ فَوْقَ ٱلْكُلِّ، 14 أَوْ لِلْوُلَاةِ فَكَمُرْسَلِينَ مِنْهُ لِلِٱنْتِقَامِ مِنْ فَاعِلِي ٱلشَّرِّ✡ رو13‏:4 ، وَلِلْمَدْحِ لِفَاعِلِي ٱلْخَيْرِ✡ رو13‏:3. 15 لِأَنَّ هَكَذَا هِيَ مَشِيئَةُ ٱللهِ: أَنْ تَفْعَلُوا ٱلْخَيْرَ فَتُسَكِّتُوا§ أو تكمّوا. انظر 1كو9‏:9 جَهَالَةَ ٱلنَّاسِ ٱلْأَغْبِيَاءِ✡ ع12؛ تي2‏:8. 16 كَأَحْرَارٍ✡ غل5‏:1، 13 ، وَلَيْسَ كَٱلَّذِينَ ٱلْحُرِّيَّةُ عِنْدَهُمْ سُتْرَةٌ لِلشَّرِّ، بَلْ كَعَبِيدِ ٱللهِ✡ 1كو7‏:22 . 17 أَكْرِمُوا ٱلْجَمِيعَ✡ رو12‏:10؛ في2‏:3 . أَحِبُّوا ٱلْإِخْوَةَ✡ ص1‏:22؛ عب13‏:1. خَافُوا ٱللهَ✡ أم24‏:21؛ مت22‏:21؛ رو13‏:7. أَكْرِمُوا ٱلْمَلِكَ.
18 أَيُّهَا ٱلْخُدَّامُ، كُونُوا خَاضِعِينَ بِكُلِّ هَيْبَةٍ لِلسَّادَةِ✡ أف6‏:5؛ كو3‏:22؛ 1تي6‏:1؛ تي2‏:9، لَيْسَ لِلصَّالِحِينَ ٱلْمُتَرَفِّقِينَ فَقَطْ، بَلْ لِلْعُنَفَاءِ* أو المعوجين أو الملتوين أَيْضًا. 19 لِأَنَّ هَذَا فَضْلٌ† ي نعمة. انظر لو6‏:32، إِنْ كَانَ أَحَدٌ مِنْ أَجْلِ ضَمِيرٍ نَحْوَ ٱللهِ، يَحْتَمِلُ أَحْزَانًا مُتَأَلِّمًا بِٱلظُّلْمِ✡ ص3‏:14؛ 4‏:14، 15؛ مت5‏:10؛ رو13‏:5. 20 لِأَنَّهُ أَيُّ مَجْدٍ هُوَ إِنْ كُنْتُمْ تُلْطَمُونَ مُخْطِئِينَ فَتَصْبِرُونَ؟ بَلْ إِنْ كُنْتُمْ تَتَأَلَّمُونَ عَامِلِينَ ٱلْخَيْرَ فَتَصْبِرُونَ، فَهَذَا فَضْلٌ عِنْدَ ٱللهِ، 21 لِأَنَّكُمْ لِهَذَا دُعِيتُمْ✡ مت16‏:24؛ أع14‏:22؛ 1تس3‏:3؛ 2تي3‏:12 . فَإِنَّ ٱلْمَسِيحَ أَيْضًا تَأَلَّمَ لِأَجْلِنَا✡ ص3‏:18، تَارِكًا لَنَا مِثَالًا لِكَيْ تَتَّبِعُوا خُطُوَاتِهِ✡ يو13‏:15؛ في2‏:5؛ 1يو2‏:6. 22 «ٱلَّذِي لَمْ يَفْعَلْ خَطِيَّةً، وَلَا وُجِدَ فِي فَمِهِ مَكْرٌ✡ إش53‏:9؛ لو23‏:41؛ يو8‏:46؛ 2كو5‏:21؛ عب4‏:15 »، 23 ٱلَّذِي إِذْ شُتِمَ لَمْ يَكُنْ يَشْتِمُ عِوَضًا✡ إش53‏:7؛ مت27‏:39؛ يو8‏:48، 49؛ عب12‏:3 ، وَإِذْ تَأَلَّمَ لَمْ يَكُنْ يُهَدِّدُ بَلْ كَانَ يُسَلِّمُ لِمَنْ يَقْضِي بِعَدْلٍ✡ لو23‏:46 . 24 ٱلَّذِي حَمَلَ هُوَ نَفْسُهُ خَطَايَانَا فِي جَسَدِهِ✡ إش53‏:4‏-6، 11؛ مت8‏:17؛ عب9‏:28 عَلَى ٱلْخَشَبَةِ، لِكَيْ نَمُوتَ عَنِ ٱلْخَطَايَا فَنَحْيَا لِلْبِرِّ✡ رو6‏:2، 11؛ 7‏:6. ٱلَّذِي بِجَلْدَتِهِ شُفِيتُمْ✡ إش53‏:5. 25 لِأَنَّكُمْ كُنْتُمْ كَخِرَافٍ ضَالَّةٍ✡ إش53‏:6؛ حز34‏:6، لَكِنَّكُمْ رَجَعْتُمُ ٱلْآنَ إِلَى رَاعِي نُفُوسِكُمْ وَأُسْقُفِهَا✡ ص5‏:4؛ حز34‏:31؛ 37‏:24؛ يو10‏:11، 14، 16؛ عب13‏:20‡ أو ناظرها. انظر أع20‏:28.
