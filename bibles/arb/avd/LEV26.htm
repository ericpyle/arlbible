مكافأة الطاعة
1 «لَا تَصْنَعُوا لَكُمْ أَوْثَانًا✡ خر20‏:4، 5؛ تث5‏:8؛ 16‏:22؛ 27‏:15؛ مز97‏:7، وَلَا تُقِيمُوا لَكُمْ تِمْثَالًا مَنْحُوتًا أَوْ نَصَبًا، وَلَا تَجْعَلُوا فِي أَرْضِكُمْ حَجَرًا مُصَوَّرًا لِتَسْجُدُوا لَهُ. لِأَنِّي أَنَا ٱلرَّبُّ إِلَهُكُمْ. 2 سُبُوتِي تَحْفَظُونَ✡ ص19‏:30 وَمَقْدِسِي تَهَابُونَ. أَنَا ٱلرَّبُّ.
3 «إِذَا سَلَكْتُمْ فِي فَرَائِضِي✡ تث11‏:13‏-15؛ 28‏:1‏-14 وَحَفِظْتُمْ وَصَايَايَ وَعَمِلْتُمْ بِهَا، 4 أُعْطِي مَطَرَكُمْ فِي حِينِهِ✡ إش30‏:23؛ حز34‏:26؛ يؤ2‏:23، 24، وَتُعْطِي ٱلْأَرْضُ غَلَّتَهَا✡ مز67‏:6؛ 85‏:12؛ حز34‏:27؛ 36‏:30؛ زك8‏:12، وَتُعْطِي أَشْجَارُ ٱلْحَقْلِ أَثْمَارَهَا، 5 وَيَلْحَقُ دِرَاسُكُمْ بِٱلْقِطَافِ✡ عا9‏:13، وَيَلْحَقُ ٱلْقِطَافُ بِٱلزَّرْعِ، فَتَأْكُلُونَ خُبْزَكُمْ لِلشِّبَعِ✡ ص25‏:19؛ تث11‏:15؛ يؤ2‏:19، 26 وَتَسْكُنُونَ فِي أَرْضِكُمْ آمِنِينَ✡ ص25‏:18؛ أي11‏:18؛ حز34‏:25، 27، 28. 6 وَأَجْعَلُ سَلَامًا فِي ٱلْأَرْضِ✡ 1أخ22‏:9؛ مز29‏:11؛ 147‏:14؛ إش45‏:7؛ حج2‏:9، فَتَنَامُونَ وَلَيْسَ مَنْ يُزْعِجُكُمْ✡ أي11‏:19؛ مز3‏:5؛ 4‏:8؛ إش35‏:9؛ إر30‏:10؛ حز34‏:25؛ هو2‏:18؛ صف 3‏:13. وَأُبِيدُ ٱلْوُحُوشَ ٱلرَّدِيئَةَ مِنَ ٱلْأَرْضِ✡ 2مل17‏:25؛ حز5‏:17؛ 14‏:15، وَلَا يَعْبُرُ سَيْفٌ فِي أَرْضِكُمْ✡ حز14‏:17. 7 وَتَطْرُدُونَ أَعْدَاءَكُمْ فَيَسْقُطُونَ أَمَامَكُمْ بِٱلسَّيْفِ. 8 يَطْرُدُ خَمْسَةٌ مِنْكُمْ مِئَةً، وَمِئَةٌ✡ تث32‏:30؛ يش23‏:10 مِنْكُمْ يَطْرُدُونَ رَبْوَةً، وَيَسْقُطُ أَعْدَاؤُكُمْ أَمَامَكُمْ بِٱلسَّيْفِ. 9 وَأَلْتَفِتُ إِلَيْكُمْ✡ خر2‏:25؛ 2مل13‏:23 وَأُثْمِرُكُمْ وَأُكَثِّرُكُمْ✡ تك17‏:6، 7؛ نح9‏:23؛ مز107‏:38 وَأَفِي مِيثَاقِي مَعَكُمْ، 10 فَتَأْكُلُونَ ٱلْعَتِيقَ ٱلْمُعَتَّقَ✡ ص25‏:22، وَتُخْرِجُونَ ٱلْعَتِيقَ مِنْ وَجْهِ* أو من أمام ٱلْجَدِيدِ. 11 وَأَجْعَلُ مَسْكَنِي فِي وَسَطِكُمْ✡ خر25‏:8؛ 29‏:45؛ يش22‏:19؛ مز76‏:2؛ حز37‏:26‏-28؛ رؤ21‏:3، وَلَا تَرْذُلُكُمْ نَفْسِي✡ ص20‏:23؛ تث32‏:19. 12 وَأَسِيرُ بَيْنَكُمْ✡ 2كو6‏:16 وَأَكُونُ لَكُمْ إِلَهًا✡ خر6‏:7؛ إر7‏:23؛ 11‏:4؛ 30‏:22؛ حز11‏:20؛ 36‏:28 وَأَنْتُمْ تَكُونُونَ لِي شَعْبًا. 13 أَنَا ٱلرَّبُّ إِلَهُكُمُ ٱلَّذِي أَخْرَجَكُمْ مِنْ أَرْضِ مِصْرَ✡ ص25‏:38، 42، 55 مِنْ كَوْنِكُمْ لَهُمْ عَبِيدًا، وَقَطَّعَ قُيُودَ نِيرِكُمْ✡ إر2‏:20؛ حز34‏:27 وَسَيَّرَكُمْ قِيَامًا.
عقوبة العصيان
14 «لَكِنْ إِنْ لَمْ تَسْمَعُوا لِي✡ تث28‏:15؛ مرا 2‏:17؛ ملا2‏:2 وَلَمْ تَعْمَلُوا كُلَّ هَذِهِ ٱلْوَصَايَا، 15 وَإِنْ رَفَضْتُمْ فَرَائِضِي✡ ع43؛ 2مل17‏:15 وَكَرِهَتْ أَنْفُسُكُمْ أَحْكَامِي، فَمَا عَمِلْتُمْ كُلَّ وَصَايَايَ، بَلْ نَكَثْتُمْ مِيثَاقِي، 16 فَإِنِّي أَعْمَلُ هَذِهِ بِكُمْ: أُسَلِّطُ عَلَيْكُمْ رُعْبًا✡ تث28‏:65‏-67؛ 32‏:25؛ إر15‏:8 وَسِلًّا✡ تث28‏:22 وَحُمَّى تُفْنِي ٱلْعَيْنَيْنِ✡ 1صم2‏:33 وَتُتْلِفُ ٱلنَّفْسَ. وَتَزْرَعُونَ بَاطِلًا زَرْعَكُمْ فَيَأْكُلُهُ أَعْدَاؤُكُمْ✡ تث28‏:33، 51؛ أي31‏:8؛ إر5‏:17؛ 12‏:13؛ مي6‏:15. 17 وَأَجْعَلُ وَجْهِي ضِدَّكُمْ✡ ص17‏:10 فَتَنْهَزِمُونَ أَمَامَ أَعْدَائِكُمْ✡ تث28‏:25؛ قض2‏:14؛ إر19‏:7، وَيَتَسَلَّطُ عَلَيْكُمْ مُبْغِضُوكُمْ✡ مز106‏:41، وَتَهْرُبُونَ وَلَيْسَ مَنْ يَطْرُدُكُمْ✡ ع36؛ مز53‏:5؛ أم28‏:1.
18 «وَإِنْ كُنْتُمْ مَعَ ذَلِكَ لَا تَسْمَعُونَ لِي، أَزِيدُ عَلَى تَأْدِيبِكُمْ سَبْعَةَ أَضْعَافٍ✡ 1صم2‏:5؛ مز119‏:164؛ أم24‏:16 حَسَبَ† أو من أجل. كذ21 خَطَايَاكُمْ، 19 فَأُحَطِّمُ فَخَارَ عِزِّكُمْ✡ إش25‏:11؛ 26‏:5؛ حز7‏:24؛ 30‏:6، وَأُصَيِّرُ سَمَاءَكُمْ كَٱلْحَدِيدِ، وَأَرْضَكُمْ كَٱلنُّحَاسِ✡ تث28‏:23، 20 فَتُفْرَغُ بَاطِلًا قُوَّتُكُمْ✡ مز127‏:1؛ إش49‏:4، وَأَرْضُكُمْ لَا تُعْطِي غَلَّتَهَا✡ تث11‏:17؛ 28‏:18؛ حج1‏:10، وَأَشْجَارُ ٱلْأَرْضِ لَا تُعْطِي أَثْمَارَهَا.
21 «وَإِنْ سَلَكْتُمْ مَعِي بِٱلْخِلَافِ، وَلَمْ تَشَاءُوا أَنْ تَسْمَعُوا لِي، أَزِيدُ عَلَيْكُمْ ضَرْبَاتٍ سَبْعَةَ أَضْعَافٍ حَسَبَ خَطَايَاكُمْ. 22 أُطْلِقُ عَلَيْكُمْ وُحُوشَ ٱلْبَرِّيَّةِ✡ تث32‏:24؛ 2مل17‏:25؛ حز5‏:17؛ 14‏:15 فَتُعْدِمُكُمُ ٱلْأَوْلَادَ‡ ع تثكلكم، وَتَقْرِضُ بَهَائِمَكُمْ، وَتُقَلِّلُكُمْ فَتُوحَشُ طُرُقُكُمْ✡ قض5‏:6؛ 2أخ15‏:5؛ إش33‏:8؛ مرا 1‏:4؛ زك7‏:14.
23 «وَإِنْ لَمْ تَتَأَدَّبُوا مِنِّي✡ إر2‏:30؛ 5‏:3؛ عا4‏:6‏-11 بِذَلِكَ، بَلْ سَلَكْتُمْ مَعِي بِٱلْخِلَافِ، 24 فَإِنِّي أَنَا أَسْلُكُ مَعَكُمْ بِٱلْخِلَافِ✡ 2صم22‏:27؛ مز18‏:26، وَأَضْرِبُكُمْ سَبْعَةَ أَضْعَافٍ حَسَبَ خَطَايَاكُمْ. 25 أَجْلِبُ عَلَيْكُمْ سَيْفًا✡ حز5‏:17؛ 6‏:3؛ 14‏:17؛ 29‏:8؛ 33‏:2 يَنْتَقِمُ نَقْمَةَ ٱلْمِيثَاقِ، فَتَجْتَمِعُونَ إِلَى مُدُنِكُمْ وَأُرْسِلُ فِي وَسَطِكُمُ ٱلْوَبَأَ✡ عد14‏:12؛ تث28‏:21؛ إر14‏:12؛ 24‏:10؛ 29‏:17، 18؛ عا4‏:10 فَتُدْفَعُونَ بِيَدِ ٱلْعَدُوِّ. 26 بِكَسْرِي لَكُمْ عَصَا ٱلْخُبْزِ✡ مز105‏:16؛ إش3‏:1؛ حز4‏:16؛ 5‏:16؛ 14‏:13. تَخْبِزُ عَشَرُ نِسَاءٍ خُبْزَكُمْ فِي تَنُّورٍ وَاحِدٍ، وَيَرْدُدْنَ خُبْزَكُمْ بِٱلْوَزْنِ، فَتَأْكُلُونَ وَلَا تَشْبَعُونَ✡ إش9‏:20؛ مي6‏:14؛ حج1‏:6.
27 «وَإِنْ كُنْتُمْ بِذَلِكَ لَا تَسْمَعُونَ لِي✡ ع21، 23 بَلْ سَلَكْتُمْ مَعِي بِٱلْخِلَافِ، 28 فَأَنَا أَسْلُكُ مَعَكُمْ بِٱلْخِلَافِ سَاخِطًا✡ إش59‏:18؛ 63‏:3؛ 66‏:15؛ إر21‏:5؛ حز5‏:13، 15؛ 8‏:18، وَأُؤَدِّبُكُمْ سَبْعَةَ أَضْعَافٍ حَسَبَ خَطَايَاكُمْ، 29 فَتَأْكُلُونَ لَحْمَ بَنِيكُمْ، وَلَحْمَ بَنَاتِكُمْ تَأْكُلُونَ✡ تث28‏:53؛ 2مل6‏:29؛ مرا 4‏:10؛ حز5‏:10. 30 وَأُخْرِبُ مُرْتَفَعَاتِكُمْ✡ 2أخ34‏:3، 4، 7؛ إش27‏:9؛ حز6‏:3‏-6، 13، وَأَقْطَعُ شَمْسَاتِكُمْ§ أي تماثيلكم للشمس، وَأُلْقِي جُثَثَكُمْ عَلَى جُثَثِ أَصْنَامِكُمْ✡ 2مل23‏:20؛ 2أخ34‏:5، وَتَرْذُلُكُمْ نَفْسِي✡ ص20‏:23؛ مز78‏:59؛ 89‏:38؛ إر14‏:19. 31 وَأُصَيِّرُ مُدُنَكُمْ خَرِبَةً✡ نح2‏:3؛ إر4‏:7؛ حز6‏:6، وَمَقَادِسَكُمْ مُوحِشَةً✡ مز74‏:7؛ مرا 1‏:10؛ حز9‏:6؛ 21‏:2، وَلَا أَشْتَمُّ رَائِحَةَ سَرُورِكُمْ. 32 وَأُوحِشُ ٱلْأَرْضَ✡ إر9‏:11؛ 25‏:11، 18 فَيَسْتَوْحِشُ مِنْهَا أَعْدَاؤُكُمُ ٱلسَّاكِنُونَ فِيهَا✡ تث28‏:37؛ 1مل9‏:8؛ إر18‏:16؛ 19‏:8؛ حز5‏:15. 33 وَأُذَرِّيكُمْ بَيْنَ ٱلْأُمَمِ✡ تث4‏:27؛ 28‏:64؛ مز44‏:11؛ إر9‏:16؛ حز12‏:15؛ 20‏:23؛ 22‏:15؛ زك7‏:14، وَأُجَرِّدُ وَرَاءَكُمُ ٱلسَّيْفَ فَتَصِيرُ أَرْضُكُمْ مُوحَشَةً، وَمُدُنُكُمْ تَصِيرُ خَرِبَةً. 34 حِينَئِذٍ تَسْتَوْفِي ٱلْأَرْضُ سُبُوتَهَا✡ 2أخ36‏:21 كُلَّ أَيَّامِ وَحْشَتِهَا وَأَنْتُمْ فِي أَرْضِ أَعْدَائِكُمْ. حِينَئِذٍ تَسْبِتُ ٱلْأَرْضُ وَتَسْتَوْفِي سُبُوتَهَا. 35 كُلَّ أَيَّامِ وَحْشَتِهَا تَسْبِتُ مَا لَمْ تَسْبِتْهُ مِنْ سُبُوتِكُمْ✡ ص25‏:2 فِي سَكَنِكُمْ عَلَيْهَا. 36 وَٱلْبَاقُونَ مِنْكُمْ أُلْقِي ٱلْجَبَانَةَ فِي قُلُوبِهِمْ✡ حز21‏:7، 12، 15 فِي أَرَاضِي أَعْدَائِهِمْ، فَيَهْزِمُهُمْ صَوْتُ وَرَقَةٍ مُنْدَفِعَةٍ✡ ع17؛ أي15‏:21؛ أم28‏:1، فَيَهْرُبُونَ كَٱلْهَرَبِ مِنَ ٱلسَّيْفِ، وَيَسْقُطُونَ وَلَيْسَ طَارِدٌ. 37 وَيَعْثُرُ بَعْضُهُمْ بِبَعْضٍ✡ قض7‏:22؛ 1صم14‏:15، 16؛ إش10‏:4 كَمَا مِنْ أَمَامِ ٱلسَّيْفِ وَلَيْسَ طَارِدٌ، وَلَا يَكُونُ لَكُمْ قِيَامٌ أَمَامَ أَعْدَائِكُمْ✡ يش7‏:12، 13؛ قض2‏:14، 38 فَتَهْلِكُونَ بَيْنَ ٱلشُّعُوبِ وَتَأْكُلُكُمْ أَرْضُ أَعْدَائِكُمْ. 39 وَٱلْبَاقُونَ مِنْكُمْ يَفْنَوْنَ✡ تث4‏:27؛ 28‏:65؛ نح1‏:8؛ إر3‏:25؛ 29‏:12، 13؛ حز4‏:17؛ 6‏:9؛ 20‏:43؛ 24‏:23؛ 33‏:10؛ 36‏:31؛ هو5‏:15؛ زك10‏:9 بِذُنُوبِهِمْ فِي أَرَاضِي أَعْدَائِكُمْ. وَأَيْضًا بِذُنُوبِ آبَائِهِمْ مَعَهُمْ يَفْنَوْنَ. 40 لَكِنْ إِنْ أَقَرُّوا بِذُنُوبِهِمْ✡ عد5‏:7؛ 1مل8‏:33، 35، 47؛ نح9‏:2؛ أم28‏:13؛ دا 9‏:3، 4؛ لو15‏:18؛ 1يو1‏:9 وَذُنُوبِ آبَائِهِمْ فِي خِيَانَتِهِمِ ٱلَّتِي خَانُونِي بِهَا، وَسُلُوكِهِمْ مَعِيَ ٱلَّذِي سَلَكُوا بِٱلْخِلَافِ، 41 وَإِنِّي أَيْضًا سَلَكْتُ مَعَهُمْ بِٱلْخِلَافِ وَأَتَيْتُ بِهِمْ إِلَى أَرْضِ أَعْدَائِهِمْ. إِلَّا أَنْ تَخْضَعَ✡ 1مل21‏:29؛ 2أخ12‏:6، 7، 12؛ 32‏:26؛ 33‏:12، 13 حِينَئِذٍ قُلُوبُهُمُ ٱلْغُلْفُ✡ إر6‏:10؛ 9‏:25، 26؛ حز44‏:7؛ أع7‏:51؛ رو2‏:29؛ كو2‏:11، وَيَسْتَوْفُوا حِينَئِذٍ عَنْ ذُنُوبِهِمْ، 42 أَذْكُرُ مِيثَاقِي✡ خر2‏:24؛ 6‏:5؛ مز106‏:45؛ حز16‏:60 مَعَ يَعْقُوبَ، وَأَذْكُرُ أَيْضًا مِيثَاقِي مَعَ إِسْحَاقَ، وَمِيثَاقِي مَعَ إِبْرَاهِيمَ، وَأَذْكُرُ ٱلْأَرْضَ✡ مز136‏:23. 43 وَٱلْأَرْضُ تُتْرَكُ مِنْهُمْ وَتَسْتَوْفِي سُبُوتَهَا✡ ع34، 35 فِي وَحْشَتِهَا مِنْهُمْ، وَهُمْ يَسْتَوْفُونَ عَنْ ذُنُوبِهِمْ لِأَنَّهُمْ قَدْ أَبَوْا أَحْكَامِي✡ ع15 وَكَرِهَتْ أَنْفُسُهُمْ فَرَائِضِي. 44 وَلَكِنْ مَعَ ذَلِكَ أَيْضًا مَتَى كَانُوا فِي أَرْضِ أَعْدَائِهِمْ، مَا أَبَيْتُهُمْ وَلَا كَرِهْتُهُمْ حَتَّى أُبِيدَهُمْ وَأَنْكُثَ مِيثَاقِي مَعَهُمْ✡ تث4‏:31؛ 2مل13‏:23؛ رو11‏:2، لِأَنِّي أَنَا ٱلرَّبُّ إِلَهُهُمْ. 45 بَلْ أَذْكُرُ لَهُمْ ٱلْمِيثَاقَ✡ رو11‏:28 مَعَ ٱلْأَوَّلِينَ ٱلَّذِينَ أَخْرَجْتُهُمْ مِنْ أَرْضِ مِصْرَ✡ ص22‏:33؛ 25‏:38 أَمَامَ أَعْيُنِ ٱلشُّعُوبِ✡ مز98‏:2؛ حز20‏:9، 14، 22 لِأَكُونَ لَهُمْ إِلَهًا. أَنَا ٱلرَّبُّ».
46 هَذِهِ هِيَ ٱلْفَرَائِضُ وَٱلْأَحْكَامُ وَٱلشَّرَائِعُ ٱلَّتِي وَضَعَهَا ٱلرَّبُّ بَيْنَهُ وَبَيْنَ بَنِي إِسْرَائِيلَ فِي جَبَلِ سِينَاءَ بِيَدِ مُوسَى✡ ص25‏:1؛ 27‏:34؛ تث6‏:1؛ 12‏:1؛ 33‏:4؛ يو1‏:17.
