3
لا اتكال على الجسد
1 أَخِيرًا يَا إِخْوَتِي، ٱفْرَحُوا فِي ٱلرَّبِّ✡ ص4‏:4؛ 2كو13‏:11؛ 1تس5‏:16 . كِتَابَةُ هَذِهِ ٱلْأُمُورِ إِلَيْكُمْ لَيْسَتْ عَلَيَّ ثَقِيلَةً، وَأَمَّا لَكُمْ فَهِيَ مُؤَمِّنَةٌ. 2 اُنْظُرُوا ٱلْكِلَابَ✡ إش56‏:10؛ غل5‏:15 . ٱنْظُرُوا فَعَلَةَ ٱلشَّرِّ✡ 2كو11‏:13 . ٱنْظُرُوا ٱلْقَطْعَ✡ رو2‏:28؛ غل5‏:2، 12 . 3 لِأَنَّنَا نَحْنُ ٱلْخِتَانَ✡ تث10‏:16؛ 30‏:6؛ إر4‏:4؛ رو2‏:29؛ 4‏:11، 12؛ كو2‏:11 ، ٱلَّذِينَ نَعْبُدُ ٱللهَ بِٱلرُّوحِ✡ يو4‏:23، 24؛ رو1‏:9؛ 7‏:6، وَنَفْتَخِرُ فِي ٱلْمَسِيحِ يَسُوعَ✡ غل6‏:14 ، وَلَا نَتَّكِلُ عَلَى ٱلْجَسَدِ. 4 مَعَ أَنَّ لِي أَنْ أَتَّكِلَ عَلَى ٱلْجَسَدِ✡ 2كو11‏:18، 21 أَيْضًا. إِنْ ظَنَّ وَاحِدٌ آخَرُ أَنْ يَتَّكِلَ عَلَى ٱلْجَسَدِ فَأَنَا بِٱلْأَوْلَى. 5 مِنْ جِهَةِ ٱلْخِتَانِ مَخْتُونٌ فِي ٱلْيَوْمِ ٱلثَّامِنِ✡ تك17‏:12 ، مِنْ جِنْسِ إِسْرَائِيلَ✡ 2كو11‏:22 ، مِنْ سِبْطِ بِنْيَامِينَ✡ رو11‏:1 ، عِبْرَانِيٌّ مِنَ ٱلْعِبْرَانِيِّينَ✡ 2كو11‏:22. مِنْ جِهَةِ ٱلنَّامُوسِ فَرِّيسِيٌّ✡ أع23‏:6؛ 26‏:4، 5 . 6 مِنْ جِهَةِ ٱلْغَيْرَةِ✡ أع22‏:3؛ غل1‏:13، 14 : مُضْطَهِدُ ٱلْكَنِيسَةِ✡ أع8‏:3؛ 9‏:1؛ 1كو15‏:9. مِنْ جِهَةِ ٱلْبِرِّ ٱلَّذِي فِي ٱلنَّامُوسِ✡ رو10‏:5 : بِلَا لَوْمٍ✡ لو1‏:6 . 7 لَكِنْ مَا كَانَ لِي رِبْحًا، فَهَذَا قَدْ حَسِبْتُهُ مِنْ أَجْلِ ٱلْمَسِيحِ خَسَارَةً✡ مت13‏:44 . 8 بَلْ إِنِّي أَحْسِبُ كُلَّ شَيْءٍ أَيْضًا خَسَارَةً مِنْ أَجْلِ فَضْلِ مَعْرِفَةِ ٱلْمَسِيحِ يَسُوعَ✡ إش53‏:11؛ إر9‏:23، 24؛ يو17‏:3؛ 1كو2‏:2؛ كو2‏:2 رَبِّي، ٱلَّذِي مِنْ أَجْلِهِ خَسِرْتُ كُلَّ ٱلْأَشْيَاءِ، وَأَنَا أَحْسِبُهَا نُفَايَةً لِكَيْ أَرْبَحَ ٱلْمَسِيحَ، 9 وَأُوجَدَ فِيهِ، وَلَيْسَ لِي بِرِّي ٱلَّذِي مِنَ ٱلنَّامُوسِ✡ رو10‏:3، 5 ، بَلِ ٱلَّذِي بِإِيمَانِ ٱلْمَسِيحِ✡ رو1‏:17؛ 3‏:21، 22؛ 9‏:30؛ 10‏:3، 6؛ غل2‏:16 ، ٱلْبِرُّ ٱلَّذِي مِنَ ٱللهِ بِٱلْإِيمَانِ. 10 لِأَعْرِفَهُ، وَقُوَّةَ قِيَامَتِهِ، وَشَرِكَةَ آلَامِهِ✡ رو6‏: 3‏-5؛ 8‏:17؛ 2كو4‏:10، 11؛ 2تي2‏:11، 12؛ 1بط4‏:13، مُتَشَبِّهًا بِمَوْتِهِ، 11 لَعَلِّي أَبْلُغُ إِلَى قِيَامَةِ ٱلْأَمْوَاتِ✡ أع26‏:7.
السعي نحو الهدف
12 لَيْسَ أَنِّي قَدْ نِلْتُ✡ 1تي6‏:12 أَوْ صِرْتُ كَامِلًا✡ عب12‏:23 ، وَلَكِنِّي أَسْعَى لَعَلِّي أُدْرِكُ ٱلَّذِي لِأَجْلِهِ أَدْرَكَنِي أَيْضًا ٱلْمَسِيحُ يَسُوعُ. 13 أَيُّهَا ٱلْإِخْوَةُ، أَنَا لَسْتُ أَحْسِبُ نَفْسِي أَنِّي قَدْ أَدْرَكْتُ. وَلَكِنِّي أَفْعَلُ شَيْئًا وَاحِدًا: إِذْ أَنَا أَنْسَى مَا هُوَ وَرَاءُ✡ مز45‏:10؛ لو9‏:62؛ 2كو5‏:16 وَأَمْتَدُّ إِلَى مَا هُوَ قُدَّامُ✡ 1كو9‏:24، 26؛ عب6‏:1 ، 14 أَسْعَى نَحْوَ ٱلْغَرَضِ✡ 2تي4‏:7، 8؛ عب12‏:1 لِأَجْلِ جَعَالَةِ✡ 1كو9‏:24 دَعْوَةِ ٱللهِ ٱلْعُلْيَا✡ عب3‏:1 فِي ٱلْمَسِيحِ يَسُوعَ. 15 فَلْيَفْتَكِرْ هَذَا✡ غل5‏:10 جَمِيعُ ٱلْكَامِلِينَ✡ 1كو2‏:6؛ 14‏:20 مِنَّا، وَإِنِ ٱفْتَكَرْتُمْ شَيْئًا بِخِلَافِهِ فَٱللهُ سَيُعْلِنُ لَكُمْ هَذَا أَيْضًا. 16 وَأَمَّا مَا قَدْ أَدْرَكْنَاهُ، فَلْنَسْلُكْ✡ رو12‏:16؛ 15‏:5 بِحَسَبِ ذَلِكَ ٱلْقَانُونِ✡ غل6‏:16 عَيْنِهِ، وَنَفْتَكِرْ ذَلِكَ✡ ص2‏:2 عَيْنَهُ.
17 كُونُوا مُتَمَثِّلِينَ بِي✡ ص4‏:9؛ 1كو4‏:16؛ 11‏:1؛ 1تس1‏:6 مَعًا أَيُّهَا ٱلْإِخْوَةُ، وَلَاحِظُوا ٱلَّذِينَ يَسِيرُونَ هَكَذَا كَمَا نَحْنُ عِنْدَكُمْ قُدْوَةٌ✡ 1بط5‏:3 . 18 لِأَنَّ كَثِيرِينَ يَسِيرُونَ مِمَّنْ كُنْتُ أَذْكُرُهُمْ لَكُمْ مِرَارًا، وَٱلْآنَ أَذْكُرُهُمْ أَيْضًا بَاكِيًا، وَهُمْ أَعْدَاءُ صَلِيبِ ٱلْمَسِيحِ✡ ص1‏:15، 16؛ غل1‏:7؛ 2‏:21؛ 6‏:12، 19 ٱلَّذِينَ نِهَايَتُهُمُ ٱلْهَلَاكُ✡ 2كو11‏:15؛ 2بط2‏:1 ، ٱلَّذِينَ إِلَهُهُمْ بَطْنُهُمْ✡ رو16‏:18؛ 1تي6‏:5؛ تي1‏:11 وَمَجْدُهُمْ فِي خِزْيِهِمِ✡ هو4‏:7؛ 2كو11‏:12؛ غل6‏:13 ، ٱلَّذِينَ يَفْتَكِرُونَ فِي ٱلْأَرْضِيَّاتِ✡ رو8‏:5 . 20 فَإِنَّ سِيرَتَنَا* أو المملكة أو الدولة التي نحن رعيتها هي.. إلخ نَحْنُ هِيَ فِي ٱلسَّمَاوَاتِ✡ أف2‏:6، 19؛ كو3‏:1، 3 ، ٱلَّتِي مِنْهَا✡ أع1‏:11 أَيْضًا نَنْتَظِرُ مُخَلِّصًا هُوَ ٱلرَّبُّ يَسُوعُ ٱلْمَسِيحُ✡ 1كو1‏:7؛ 1تس1‏:10؛ تي2‏:13 ، 21 ٱلَّذِي سَيُغَيِّرُ شَكْلَ جَسَدِ تَوَاضُعِنَا✡ 1كو15‏:43، 48، 49؛ كو3‏:4؛ 1يو3‏:2 لِيَكُونَ عَلَى صُورَةِ جَسَدِ مَجْدِهِ، بِحَسَبِ عَمَلِ ٱسْتِطَاعَتِهِ✡ أف1‏:19 أَنْ يُخْضِعَ لِنَفْسِهِ كُلَّ شَيْءٍ✡ 1كو15‏:26، 27.
