الأحياء مع المسيح
1 وَأَنْتُمْ إِذْ كُنْتُمْ أَمْوَاتًا بِٱلذُّنُوبِ وَٱلْخَطَايَا✡ ع5؛ ص4‏:18؛ كو2‏:13، 2 ٱلَّتِي سَلَكْتُمْ فِيهَا قَبْلًا✡ ص4‏:22؛ 1كو6‏:11؛ كو1‏:21؛ 3‏:7؛ 1يو5‏:19 حَسَبَ دَهْرِ هَذَا ٱلْعَالَمِ، حَسَبَ رَئِيسِ سُلْطَانِ ٱلْهَوَاءِ✡ ص6‏:12 ، ٱلرُّوحِ ٱلَّذِي يَعْمَلُ ٱلْآنَ فِي أَبْنَاءِ ٱلْمَعْصِيَةِ✡ ص5‏:6؛ كو3‏:6، 3 ٱلَّذِينَ نَحْنُ أَيْضًا جَمِيعًا تَصَرَّفْنَا قَبْلًا✡ تي3‏:3؛ 1بط4‏:3 بَيْنَهُمْ فِي شَهَوَاتِ جَسَدِنَا✡ غل5‏:16 ، عَامِلِينَ مَشِيئَاتِ ٱلْجَسَدِ وَٱلْأَفْكَارِ، وَكُنَّا بِٱلطَّبِيعَةِ أَبْنَاءَ ٱلْغَضَبِ✡ مز51‏:5؛ رو5‏:12، 14 كَٱلْبَاقِينَ أَيْضًا، 4 ٱللهُ ٱلَّذِي هُوَ غَنِيٌّ فِي ٱلرَّحْمَةِ✡ ع7؛ ص1‏:7؛ رو10‏:12، مِنْ أَجْلِ مَحَبَّتِهِ ٱلْكَثِيرَةِ ٱلَّتِي أَحَبَّنَا بِهَا، 5 وَنَحْنُ أَمْوَاتٌ بِٱلْخَطَايَا✡ ع1؛ رو5‏:6، 8، 10 أَحْيَانَا مَعَ ٱلْمَسِيحِ✡ يو5‏:24؛ رو6‏:4، 5؛ كو2‏:12، 13؛ 3‏:1، 3 -بِٱلنِّعْمَةِ أَنْتُمْ مُخَلَّصُونَ✡ ع8؛ أع15‏:11؛ تي3‏:5- 6 وَأَقَامَنَا مَعَهُ، وَأَجْلَسَنَا مَعَهُ فِي ٱلسَّمَاوِيَّاتِ✡ ص1‏:20 فِي ٱلْمَسِيحِ يَسُوعَ، 7 لِيُظْهِرَ فِي ٱلدُّهُورِ ٱلْآتِيَةِ غِنَى نِعْمَتِهِ ٱلْفَائِقَ، بِٱللُّطْفِ✡ تي3‏:4 عَلَيْنَا فِي ٱلْمَسِيحِ يَسُوعَ. 8 لِأَنَّكُمْ بِٱلنِّعْمَةِ مُخَلَّصُونَ✡ ع5؛ رو3‏:24؛ 2تي1‏:9، بِٱلْإِيمَانِ✡ رو4‏:16 ، وَذَلِكَ لَيْسَ مِنْكُمْ. هُوَ عَطِيَّةُ ٱللهِ✡ ص1‏:19؛ مت16‏:17؛ يو6‏:44، 65؛ رو10‏:14، 15، 17؛ في1‏:29. 9 لَيْسَ مِنْ أَعْمَالٍ كَيْلَا يَفْتَخِرَ أَحَدٌ✡ رو3‏:20، 27، 28؛ 4‏:2؛ 9‏:11؛ 11‏:6؛ 1كو1‏:29‏-31؛ 2تي1‏:9؛ تي3‏:5. 10 لِأَنَّنَا نَحْنُ عَمَلُهُ✡ ص4‏:24؛ تث32‏:6؛ مز100‏:3؛ إش19‏:25؛ 29‏:23؛ 44‏:21؛ يو3‏:3، 5؛ 1كو3‏:9؛ 2كو5‏:5، 17؛ تي2‏:14، مَخْلُوقِينَ فِي ٱلْمَسِيحِ يَسُوعَ لِأَعْمَالٍ صَالِحَةٍ، قَدْ سَبَقَ ٱللهُ فَأَعَدَّهَا لِكَيْ نَسْلُكَ فِيهَا✡ ص1‏:4 .
واحد في المسيح
11 لِذَلِكَ ٱذْكُرُوا أَنَّكُمْ أَنْتُمُ ٱلْأُمَمُ قَبْلًا✡ ص5‏:8؛ 1كو12‏:2؛ كو1‏:21؛ 2‏:13 فِي ٱلْجَسَدِ، ٱلْمَدْعُوِّينَ غُرْلَةً مِنَ ٱلْمَدْعُوِّ خِتَانًا✡ رو2‏:28، 29؛ كو2‏:11 مَصْنُوعًا بِٱلْيَدِ فِي ٱلْجَسَدِ، 12 أَنَّكُمْ كُنْتُمْ فِي ذَلِكَ ٱلْوَقْتِ بِدُونِ مَسِيحٍ✡ ص4‏:18؛ كو1‏:21 ، أَجْنَبِيِّينَ✡ حز13‏:9؛ يو10‏:16 عَنْ رَعَوِيَّةِ✡ أع22‏:28 إِسْرَائِيلَ، وَغُرَبَاءَ عَنْ عُهُودِ ٱلْمَوْعِدِ✡ رو9‏:4، 5 ، لَا رَجَاءَ لَكُمْ✡ 1تس4‏:13 ، وَبِلَا إِلَهٍ✡ غل4‏:8؛ 1تس4‏:5 فِي ٱلْعَالَمِ. 13 وَلَكِنِ ٱلْآنَ فِي ٱلْمَسِيحِ يَسُوعَ، أَنْتُمُ ٱلَّذِينَ كُنْتُمْ قَبْلًا بَعِيدِينَ، صِرْتُمْ قَرِيبِينَ بِدَمِ ٱلْمَسِيحِ✡ ع17؛ أع2‏:39؛ غل3‏:28. 14 لِأَنَّهُ هُوَ سَلَامُنَا✡ مي5‏:5؛ يو16‏:33؛ أع10‏:36؛ رو5‏:1؛ كو1‏:20، ٱلَّذِي جَعَلَ ٱلِٱثْنَيْنِ وَاحِدًا✡ يو10‏:16؛ غل3‏:28، وَنَقَضَ حَائِطَ ٱلسِّيَاجِ ٱلْمُتَوَسِّطَ، 15 أَيِ ٱلْعَدَاوَةَ. مُبْطِلًا بِجَسَدِهِ✡ كو1‏:22؛ 2‏:14، 20 نَامُوسَ ٱلْوَصَايَا فِي فَرَائِضَ، لِكَيْ يَخْلُقَ ٱلِٱثْنَيْنِ فِي نَفْسِهِ إِنْسَانًا وَاحِدًا جَدِيدًا✡ ص4‏:24؛ 2كو5‏:17؛ غل6‏:15، صَانِعًا سَلَامًا، 16 وَيُصَالِحَ ٱلِٱثْنَيْنِ فِي جَسَدٍ وَاحِدٍ مَعَ ٱللهِ✡ كو1‏:20‏-22 بِٱلصَّلِيبِ، قَاتِلًا ٱلْعَدَاوَةَ بِهِ✡ رو6‏:6؛ 8‏:3؛ كو2‏:14 . 17 فَجَاءَ وَبَشَّرَكُمْ بِسَلَامٍ✡ ع13، 14؛ إش57‏:19؛ زك9‏:10؛ أع2‏:39؛ 10‏:36؛ رو5‏:1، أَنْتُمُ ٱلْبَعِيدِينَ وَٱلْقَرِيبِينَ✡ مز148‏:14؛ أع2‏:39. 18 لِأَنَّ بِهِ لَنَا كِلَيْنَا قُدُومًا✡ ص3‏:12؛ يو10‏:9؛ 14‏:6؛ رو5‏:2؛ عب4‏:16؛ 10‏:19، 20؛ 1بط3‏:18 فِي رُوحٍ وَاحِدٍ✡ ص4‏:4؛ 1كو12‏:13 إِلَى ٱلْآبِ. 19 فَلَسْتُمْ إِذًا بَعْدُ غُرَبَاءَ وَنُزُلًا✡ تك23‏:4 ، بَلْ رَعِيَّةٌ مَعَ ٱلْقِدِّيسِينَ✡ في3‏:20؛ عب12‏:22، 23 وَأَهْلِ بَيْتِ ٱللهِ✡ ص3‏:15؛ غل6‏:10، 20 مَبْنِيِّينَ✡ ص4‏:12؛ 1كو3‏:9، 10؛ 1بط2‏:4، 5 عَلَى أَسَاسِ✡ مت16‏:18؛ غل2‏:9؛ رؤ21‏:14 ٱلرُّسُلِ وَٱلْأَنْبِيَاءِ✡ ص4‏:11؛ 1كو12‏:28، وَيَسُوعُ ٱلْمَسِيحُ نَفْسُهُ حَجَرُ ٱلزَّاوِيَةِ✡ مز118‏:22؛ إش28‏:16؛ مت21‏:42 ، 21 ٱلَّذِي فِيهِ كُلُّ ٱلْبِنَاءِ مُرَكَّبًا مَعًا✡ ص4‏:15، 16 ، يَنْمُو هَيْكَلًا مُقَدَّسًا فِي ٱلرَّبِّ✡ 1كو3‏:17؛ 6‏:19؛ 2كو6‏:16 . 22 ٱلَّذِي فِيهِ أَنْتُمْ أَيْضًا مَبْنِيُّونَ مَعًا، مَسْكَنًا لِلهِ فِي ٱلرُّوحِ✡ 1بط2‏:5 .
