السلوك المسيحي
1 فَإِنْ كُنْتُمْ قَدْ قُمْتُمْ مَعَ ٱلْمَسِيحِ✡ ص2‏:12؛ رو6‏:5؛ أف2‏:6 فَٱطْلُبُوا مَا فَوْقُ، حَيْثُ ٱلْمَسِيحُ جَالِسٌ عَنْ يَمِينِ ٱللهِ✡ رو8‏:34؛ أف1‏:20. 2 ٱهْتَمُّوا بِمَا فَوْقُ لَا بِمَا عَلَى ٱلْأَرْضِ، 3 لِأَنَّكُمْ قَدْ مُتُّمْ✡ ص2‏:20؛ رو6‏:2؛ غل2‏:20 وَحَيَاتُكُمْ مُسْتَتِرَةٌ✡ ص1‏:5؛ 2كو5‏:7 مَعَ ٱلْمَسِيحِ فِي ٱللهِ. 4 مَتَى أُظْهِرَ ٱلْمَسِيحُ✡ 1يو3‏:2 حَيَاتُنَا✡ يو11‏:25؛ 14‏:6، فَحِينَئِذٍ تُظْهَرُونَ أَنْتُمْ أَيْضًا مَعَهُ فِي ٱلْمَجْدِ✡ 1كو15‏:43؛ في3‏:21 .
5 فَأَمِيتُوا✡ رو8‏:13؛ غل5‏:24 أَعْضَاءَكُمُ✡ رو6‏:13 ٱلَّتِي عَلَى ٱلْأَرْضِ: ٱلزِّنَا✡ أف5‏:3 ، ٱلنَّجَاسَةَ، ٱلْهَوَى✡ رو1‏:26 ، ٱلشَّهْوَةَ ٱلرَّدِيَّةَ✡ 1تس4‏:5، ٱلطَّمَعَ -ٱلَّذِي* أو فأنه هُوَ عِبَادَةُ ٱلْأَوْثَانِ✡ أف5‏:5 - 6 ٱلْأُمُورَ ٱلَّتِي مِنْ أَجْلِهَا† أو الذي من أجله يَأْتِي غَضَبُ ٱللهِ✡ رو1‏:18؛ أف5‏:6؛ رؤ22‏:15 عَلَى أَبْنَاءِ ٱلْمَعْصِيَةِ✡ أف2‏:2؛ 2بط2‏:14 ، 7 ٱلَّذِينَ بَيْنَهُمْ‡ أو التي فيها أَنْتُمْ أَيْضًا سَلَكْتُمْ✡ ص2‏:13؛ رو6‏:19، 20؛ 7‏:5؛ 1كو6‏:11؛ أف2‏:2؛ تي3‏:3 قَبْلًا، حِينَ كُنْتُمْ تَعِيشُونَ فِيهَا. 8 وَأَمَّا ٱلْآنَ فَٱطْرَحُوا✡ أف4‏:22؛ عب12‏:1؛ يع1‏:21؛ 1بط2‏:1 عَنْكُمْ أَنْتُمْ أَيْضًا ٱلْكُلَّ: ٱلْغَضَبَ، ٱلسَّخَطَ، ٱلْخُبْثَ، ٱلتَّجْدِيفَ، ٱلْكَلَامَ ٱلْقَبِيحَ مِنْ أَفْوَاهِكُمْ✡ أف4‏:29؛ 5‏:4. 9 لَا تَكْذِبُوا بَعْضُكُمْ عَلَى بَعْضٍ✡ لا19‏:11؛ أف4‏:25 ، إِذْ خَلَعْتُمُ§ أو خالعين ٱلْإِنْسَانَ ٱلْعَتِيقَ✡ أف4‏:22، 24 مَعَ أَعْمَالِهِ، 10 وَلَبِسْتُمُ* أو لابسين ٱلْجَدِيدَ ٱلَّذِي يَتَجَدَّدُ✡ رو12‏:2 لِلْمَعْرِفَةِ حَسَبَ صُورَةِ خَالِقِهِ✡ أف2‏:10؛ 4‏:23، 24 ، 11 حَيْثُ لَيْسَ يُونَانِيٌّ✡ رو10‏:12؛ 1كو12‏:13؛ غل3‏:28؛ 5‏:6؛ أف6‏:8 وَيَهُودِيٌّ، خِتَانٌ وَغُرْلَةٌ، بَرْبَرِيٌّ سِكِّيثِيٌّ† أو تتر، عَبْدٌ حُرٌّ، بَلِ ٱلْمَسِيحُ ٱلْكُلُّ وَفِي ٱلْكُلِّ✡ أف1‏:23 .
12 فَٱلْبَسُوا✡ أف4‏:24 كَمُخْتَارِي ٱللهِ✡ 1تس1‏:4؛ 1بط1‏:2؛ 2بط1‏:10 ٱلْقِدِّيسِينَ ٱلْمَحْبُوبِينَ أَحْشَاءَ رَأْفَاتٍ✡ غل5‏:22؛ أف4‏:2، 32؛ في2‏:1 ، وَلُطْفًا، وَتَوَاضُعًا، وَوَدَاعَةً، وَطُولَ أَنَاةٍ، 13 مُحْتَمِلِينَ بَعْضُكُمْ بَعْضًا✡ مر11‏:25؛ أف4‏:2، 32 ، وَمُسَامِحِينَ بَعْضُكُمْ بَعْضًا. إِنْ كَانَ لِأَحَدٍ عَلَى أَحَدٍ شَكْوَى، كَمَا غَفَرَ لَكُمُ ٱلْمَسِيحُ هَكَذَا أَنْتُمْ أَيْضًا. 14 وَعَلَى جَمِيعِ هَذِهِ✡ 1بط4‏:8 ٱلْبَسُوا ٱلْمَحَبَّةَ✡ ص2‏:2؛ يو13‏:34؛ رو13‏:8؛ 1كو13؛ أف5‏:2؛ 1تس4‏:9؛ 1تي1‏:5؛ 1يو3‏:23؛ 4‏:21 ٱلَّتِي هِيَ‡ أو فإنها رِبَاطُ ٱلْكَمَالِ✡ أف4‏:3 . 15 وَلْيَمْلِكْ فِي قُلُوبِكُمْ سَلَامُ ٱللهِ✡ رو14‏:17؛ في4‏:7 ٱلَّذِي إِلَيْهِ دُعِيتُمْ✡ 1كو7‏:15 فِي جَسَدٍ وَاحِدٍ✡ أف2‏:16، 17؛ 4‏:4 ، وَكُونُوا شَاكِرِينَ✡ ع17؛ ص2‏:7.
16 لِتَسْكُنْ فِيكُمْ كَلِمَةُ ٱلْمَسِيحِ بِغِنًى، وَأَنْتُمْ بِكُلِّ حِكْمَةٍ مُعَلِّمُونَ وَمُنْذِرُونَ بَعْضُكُمْ بَعْضًا، بِمَزَامِيرَ✡ 1كو14‏:26؛ أف5‏:19 وَتَسَابِيحَ وَأَغَانِيَّ رُوحِيَّةٍ، بِنِعْمَةٍ✡ ص4‏:6، مُتَرَنِّمِينَ فِي قُلُوبِكُمْ لِلرَّبِّ. 17 وَكُلُّ مَا عَمِلْتُمْ بِقَوْلٍ أَوْ فِعْلٍ، فَٱعْمَلُوا ٱلْكُلَّ بِٱسْمِ ٱلرَّبِّ يَسُوعَ✡ 1كو10‏:31 ، شَاكِرِينَ ٱللهَ وَٱلْآبَ بِهِ✡ ص1‏:12؛ 2‏:7؛ رو1‏:8؛ أف5‏:20؛ 1تس5‏:18؛ عب13‏:15.
وصايا للبيت المسيحي
18 أَيَّتُهَا ٱلنِّسَاءُ، ٱخْضَعْنَ لِرِجَالِكُنَّ✡ أف5‏:22؛ تي2‏:5؛ 1بط3‏:1 كَمَا يَلِيقُ فِي ٱلرَّبِّ✡ أع5‏:29؛ أف5‏:3؛ 6‏:1 . 19 أَيُّهَا ٱلرِّجَالُ، أَحِبُّوا نِسَاءَكُمْ✡ أف5‏:25، 28، 29، 33؛ 1بط3‏:7، وَلَا تَكُونُوا قُسَاةً§ ي مرّين عَلَيْهِنَّ✡ أف4‏:31 . 20 أَيُّهَا ٱلْأَوْلَادُ، أَطِيعُوا وَالِدِيكُمْ✡ أف6‏:1 فِي كُلِّ شَيْءٍ✡ أف5‏:24؛ تي2‏:9 لِأَنَّ هَذَا مَرْضِيٌّ فِي ٱلرَّبِّ. 21 أَيُّهَا ٱلْآبَاءُ، لَا تُغِيظُوا أَوْلَادَكُمْ✡ أف6‏:4 لِئَلَّا يَفْشَلُوا. 22 أَيُّهَا ٱلْعَبِيدُ✡ أف6‏:5‏-8؛ 1تي6‏:1؛ تي2‏:9؛ 1بط2‏:18، أَطِيعُوا فِي كُلِّ شَيْءٍ✡ ع20 سَادَتَكُمْ حَسَبَ ٱلْجَسَدِ✡ فل16 ، لَا بِخِدْمَةِ ٱلْعَيْنِ كَمَنْ يُرْضِي ٱلنَّاسَ، بَلْ بِبَسَاطَةِ ٱلْقَلْبِ، خَائِفِينَ ٱلرَّبَّ. 23 وَكُلُّ مَا فَعَلْتُمْ، فَٱعْمَلُوا مِنَ ٱلْقَلْبِ* ي النفس، كَمَا لِلرَّبِّ لَيْسَ لِلنَّاسِ✡ أف6‏:6، 7، 24 عَالِمِينَ أَنَّكُمْ مِنَ ٱلرَّبِّ سَتَأْخُذُونَ جَزَاءَ ٱلْمِيرَاثِ✡ أف6‏:8، لِأَنَّكُمْ تَخْدِمُونَ ٱلرَّبَّ ٱلْمَسِيحَ✡ 1كو7‏:22 . 25 وَأَمَّا ٱلظَّالِمُ فَسَينَالُ مَا ظَلَمَ بِهِ، وَلَيْسَ مُحَابَاةٌ✡ تث10‏:17؛ رو2‏:11؛ أف6‏:9؛ 1بط1‏:17.
