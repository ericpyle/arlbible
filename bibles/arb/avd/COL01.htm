رِسَالَةُ بُولُسَ ٱلرَّسُولِ إِلَى أَهْلِ كُولُوسِّي
1
1 بُولُسُ، رَسُولُ يَسُوعَ ٱلْمَسِيحِ✡ أف1‏:1 بِمَشِيئَةِ ٱللهِ، وَتِيمُوثَاوُسُ ٱلْأَخُ، 2 إِلَى ٱلْقِدِّيسِينَ فِي كُولُوسِّي، وَٱلْإِخْوَةِ ٱلْمُؤْمِنِينَ✡ 1كو4‏:17؛ أف6‏:21 فِي ٱلْمَسِيحِ: نِعْمَةٌ لَكُمْ وَسَلَامٌ مِنَ ٱللهِ أَبِينَا وَٱلرَّبِّ يَسُوعَ ٱلْمَسِيحِ✡ غل1‏:3.
شكر وصلاة
3 نَشْكُرُ ٱللهَ وَأَبَا رَبِّنَا يَسُوعَ ٱلْمَسِيحِ كُلَّ حِينٍ، مُصَلِّينَ لِأَجْلِكُمْ✡ 1كو1‏:4؛ أف1‏:16؛ في1‏:3؛ 4‏:6، 4 إِذْ سَمِعْنَا إِيمَانَكُمْ✡ ع9؛ أف1‏:15؛ فل5 بِٱلْمَسِيحِ يَسُوعَ، وَمَحَبَّتَكُمْ لِجَمِيعِ ٱلْقِدِّيسِينَ✡ عب6‏:10، 5 مِنْ أَجْلِ ٱلرَّجَاءِ ٱلْمَوْضُوعِ لَكُمْ فِي ٱلسَّمَاوَاتِ✡ 2تي4‏:8؛ 1بط1‏:4 ، ٱلَّذِي سَمِعْتُمْ بِهِ قَبْلًا فِي كَلِمَةِ حَقِّ ٱلْإِنْجِيلِ، 6 ٱلَّذِي قَدْ حَضَرَ إِلَيْكُمْ كَمَا فِي كُلِّ ٱلْعَالَمِ✡ ع23؛ مت24‏:14؛ مر16‏:15؛ رو10‏:18 أَيْضًا، وَهُوَ مُثْمِرٌ✡ مر4‏:8؛ يو15‏:16؛ في1‏:11 كَمَا فِيكُمْ أَيْضًا مُنْذُ يَوْمَ سَمِعْتُمْ وَعَرَفْتُمْ نِعْمَةَ ٱللهِ بِٱلْحَقِيقَةِ✡ 2كو6‏:1؛ أف3‏:2؛ تي2‏:11؛ 1بط5‏:12 . 7 كَمَا تَعَلَّمْتُمْ أَيْضًا مِنْ أَبَفْرَاسَ✡ ص4‏:12؛ فل23 ٱلْعَبْدِ ٱلْحَبِيبِ مَعَنَا، ٱلَّذِي هُوَ خَادِمٌ أَمِينٌ✡ 2كو11‏:23؛ 1تي4‏:6 لِلْمَسِيحِ لِأَجْلِكُمُ، 8 ٱلَّذِي أَخْبَرَنَا أَيْضًا بِمَحَبَّتِكُمْ فِي ٱلرُّوحِ✡ رو15‏:30. 9 مِنْ أَجْلِ ذَلِكَ نَحْنُ أَيْضًا، مُنْذُ يَوْمَ سَمِعْنَا✡ ع3، 4؛ أف1‏:15، 16، لَمْ نَزَلْ مُصَلِّينَ وَطَالِبِينَ لِأَجْلِكُمْ أَنْ تَمْتَلِئُوا✡ 1كو1‏:5 مِنْ مَعْرِفَةِ مَشِيئَتِهِ✡ رو12‏:2؛ أف5‏:10، 17، فِي كُلِّ حِكْمَةٍ✡ أف1‏:8 وَفَهْمٍ رُوحِيٍّ، 10 لِتَسْلُكُوا كَمَا يَحِقُّ✡ أف4‏:1؛ في1‏:27؛ 1تس2‏:12 لِلرَّبِّ، فِي كُلِّ رِضًى✡ 1تس4‏:1 ، مُثْمِرِينَ فِي كُلِّ عَمَلٍ صَالِحٍ✡ يو15‏:16؛ 2كو9‏:8؛ في1‏:11؛ تي3‏:1؛ عب13‏:21 ، وَنَامِينَ فِي مَعْرِفَةِ ٱللهِ، 11 مُتَقَوِّينَ بِكُلِّ قُوَّةٍ✡ أف3‏:16؛ 6‏:10 بِحَسَبِ قُدْرَةِ مَجْدِهِ، لِكُلِّ صَبْرٍ✡ أف4‏:2 وَطُولِ أَنَاةٍ بِفَرَحٍ✡ أع5‏:41؛ رو5‏:3 ، 12 شَاكِرِينَ ٱلْآبَ✡ ص3‏:15؛ أف5‏:20 ٱلَّذِي أَهَّلَنَا* أو جعلنا كفاة لِشَرِكَةِ مِيرَاثِ ٱلْقِدِّيسِينَ فِي ٱلنُّورِ✡ أع26‏:18؛ أف1‏:11 ، 13 ٱلَّذِي أَنْقَذَنَا مِنْ سُلْطَانِ ٱلظُّلْمَةِ✡ أف6‏:12؛ عب2‏:14؛ 1بط2‏:9، وَنَقَلَنَا إِلَى مَلَكُوتِ✡ 1تس2‏:12؛ 2بط1‏:11 ٱبْنِ مَحَبَّتِهِ✡ مت3‏:17؛ أف1‏:6 ، 14 ٱلَّذِي لَنَا فِيهِ ٱلْفِدَاءُ، بِدَمِهِ غُفْرَانُ ٱلْخَطَايَا✡ أف1‏:7.
عظمة المسيح وسموه
15 ٱلَّذِي هُوَ صُورَةُ ٱللهِ غَيْرِ ٱلْمَنْظُورِ✡ 2كو4‏:4؛ عب1‏:3 ، بِكْرُ كُلِّ خَلِيقَةٍ✡ رؤ3‏:14. 16 فَإِنَّهُ فِيهِ خُلِقَ ٱلْكُلُّ✡ يو1‏:3؛ 1كو8‏:6؛ أف3‏:9؛ عب1‏:2 : مَا في ٱلسَّمَاوَاتِ وَمَا عَلَى ٱلْأَرْضِ، مَا يُرَى وَمَا لَا يُرَى، سَوَاءٌ كَانَ عُرُوشًا أَمْ سِيَادَاتٍ✡ ص2‏:10، 15؛ رو8‏:38؛ أف1‏:21؛ 1بط3‏:22 أَمْ رِيَاسَاتٍ أَمْ سَلَاطِينَ. ٱلْكُلُّ بِهِ✡ رو11‏:36؛ عب2‏:10 وَلَهُ قَدْ خُلِقَ. 17 ٱلَّذِي هُوَ قَبْلَ كُلِّ شَيْءٍ✡ يو1‏:1، 3؛ 17‏:5؛ 1كو8‏:6، وَفِيهِ يَقُومُ ٱلْكُلُّ، 18 وَهُوَ رَأْسُ ٱلْجَسَدِ: ٱلْكَنِيسَةِ✡ 1كو11‏:3؛ أف1‏:10، 22؛ 4‏:15؛ 5‏:23 . ٱلَّذِي هُوَ ٱلْبَدَاءَةُ، بِكْرٌ مِنَ ٱلْأَمْوَاتِ✡ أع26‏:23؛ 1كو15‏:20، 23؛ رؤ1‏:5 ، لِكَيْ يَكُونَ هُوَ مُتَقَدِّمًا فِي كُلِّ شَيْءٍ. 19 لِأَنَّهُ فِيهِ سُرَّ أَنْ يَحِلَّ كُلُّ ٱلْمِلْءِ✡ ص2‏:9؛ 3‏:11؛ يو1‏:16؛ 3‏:34، 20 وَأَنْ يُصَالِحَ✡ 2كو5‏:18 بِهِ ٱلْكُلَّ لِنَفْسِهِ✡ أف1‏:10، عَامِلًا ٱلصُّلْحَ✡ أف2‏:14‏-16 بِدَمِ صَلِيبِهِ، بِوَاسِطَتِهِ، سَوَاءٌ كَانَ: مَا عَلَى ٱلْأَرْضِ، أَمْ مَا فِي ٱلسَّمَاوَاتِ.
21 وَأَنْتُمُ ٱلَّذِينَ كُنْتُمْ قَبْلًا أَجْنَبِيِّينَ وَأَعْدَاءً✡ أف2‏:1، 2، 12، 19؛ 4‏:18 فِي ٱلْفِكْرِ، فِي ٱلْأَعْمَالِ ٱلشِّرِّيرَةِ✡ تي1‏:15، 16 ، قَدْ صَالَحَكُمُ ٱلْآنَ 22 فِي جِسْمِ بَشَرِيَّتِهِ✡ أف2‏:15، 16 بِٱلْمَوْتِ، لِيُحْضِرَكُمْ قِدِّيسِينَ وَبِلَا لَوْمٍ وَلَا شَكْوَى أَمَامَهُ✡ لو1‏:75؛ أف1‏:4؛ 5‏:27؛ 1تس4‏:7؛ تي2‏:14؛ يه24 ، 23 إِنْ ثَبَتُّمْ عَلَى ٱلْإِيمَانِ، مُتَأَسِّسِينَ✡ ص2‏:7؛ أف3‏:17 وَرَاسِخِينَ وَغَيْرَ مُنْتَقِلِينَ✡ يو15‏:6 عَنْ رَجَاءِ ٱلْإِنْجِيلِ، ٱلَّذِي سَمِعْتُمُوهُ، ٱلْمَكْرُوزِ بِهِ✡ رو10‏:18 فِي كُلِّ ٱلْخَلِيقَةِ✡ ع6 ٱلَّتِي تَحْتَ ٱلسَّمَاءِ، ٱلَّذِي صِرْتُ أَنَا بُولُسَ خَادِمًا لَهُ✡ ع25؛ أع1‏:17؛ 2كو3‏:6؛ 4‏:1؛ 5‏:18؛ أف3‏:7؛ 1تي2‏:7.
جهاد بولس من أجل الكنيسة
24 ٱلَّذِي ٱلْآنَ أَفْرَحُ فِي آلَامِي✡ رو5‏:3؛ 2كو7‏:4 لِأَجْلِكُمْ✡ أف3‏:1، 13 ، وَأُكَمِّلُ نَقَائِصَ شَدَائِدِ ٱلْمَسِيحِ✡ 2كو1‏:5، 6؛ في3‏:10؛ 2تي1‏:8؛ 2‏:10 فِي جِسْمِي لِأَجْلِ جَسَدِهِ، ٱلَّذِي هُوَ ٱلْكَنِيسَةُ✡ أف1‏:22، 23، 25 ٱلَّتِي صِرْتُ أَنَا خَادِمًا لَهَا، حَسَبَ تَدْبِيرِ ٱللهِ ٱلْمُعْطَى لِي لِأَجْلِكُمْ، لِتَتْمِيمِ كَلِمَةِ ٱللهِ✡ ع23؛ رو15‏:19؛ 1كو9‏:17؛ غل2‏:7؛ أف3‏:2. 26 ٱلسِّرِّ ٱلْمَكْتُومِ✡ رو16‏:25؛ 1كو2‏:7؛ أف3‏:9 مُنْذُ ٱلدُّهُورِ وَمُنْذُ ٱلْأَجْيَالِ، لَكِنَّهُ ٱلْآنَ قَدْ أُظْهِرَ لِقِدِّيسِيهِ✡ مت13‏:11؛ 2تي1‏:10 ، 27 ٱلَّذِينَ أَرَادَ ٱللهُ أَنْ يُعَرِّفَهُمْ✡ 2كو2‏:14 مَا هُوَ غِنَى مَجْدِ هَذَا ٱلسِّرِّ✡ رو9‏:23؛ أف1‏:7؛ 3‏:8 فِي ٱلْأُمَمِ، ٱلَّذِي هُوَ ٱلْمَسِيحُ فِيكُمْ رَجَاءُ ٱلْمَجْدِ✡ 1تي1‏:1 . 28 ٱلَّذِي نُنَادِي بِهِ مُنْذِرِينَ✡ أع20‏:20، 27، 31 كُلَّ إِنْسَانٍ، وَمُعَلِّمِينَ كُلَّ إِنْسَانٍ، بِكُلِّ حِكْمَةٍ، لِكَيْ نُحْضِرَ كُلَّ إِنْسَانٍ كَامِلًا فِي✡ ع22؛ 2كو11‏:2؛ أف5‏:27 ٱلْمَسِيحِ يَسُوعَ. 29 ٱلْأَمْرُ ٱلَّذِي لِأَجْلِهِ أَتْعَبُ✡ 1كو15‏:10 أَيْضًا مُجَاهِدًا✡ ص2‏:1 ، بِحَسَبِ عَمَلِهِ ٱلَّذِي يَعْمَلُ فِيَّ بِقُوَّةٍ✡ أف1‏:19؛ 3‏:7، 20.
